#include <iostream>

using namespace std;

int main(){

int dogYears = 0;
  int humanYears;

  do{
  cout << "How old is your dog?: ";
  cin >> dogYears;
  } while(dogYears < 1);

  if(dogYears == 1){
    humanYears = 15;
  }else if(dogYears == 2){
    humanYears = 24;
  }else{
    humanYears = 24 + (dogYears - 2) * 5;
  }

  if(humanYears > 80){
    cout << "Damn! " << humanYears << " years old!?" << endl;
  } else {
    cout << "Your dog's age in human years is: " << humanYears << "!" << endl;
  }


  return 0;
}
